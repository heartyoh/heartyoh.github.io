{% highlight javascript %}
function shuffle(a) {
  for(let i = a.length; i; i--) {
    let j = Math.floor(Math.random() * i);
    [a[i - 1], a[j]] = [a[j], a[i - 1]];
  }
}
var 병헌 = ['지금', '나랑', '장난', '하냐'];
shuffle(병헌);
console.error(병헌.join(' '));
